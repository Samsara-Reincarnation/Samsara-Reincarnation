actor "Slow Fire Protector": InstantSwitchWeapon
{
    Weapon.SelectionOrder 1000
    Weapon.SlotNumber 2
	Weapon.BobStyle Alpha
	Weapon.BobRangeX 0
	Weapon.BobRangeY 0.6
	Weapon.BobSpeed 1.5	
    Inventory.PickUpSound "wolfen/knifeget"
    Obituary "%k ENERGIZED! %o with the Slow Fire Protector."
    Inventory.PickupMessage "You got a Slow Fire Protector!"
    Weapon.AmmoUse 1
    Weapon.AmmoGive 10
	Weapon.AmmoType "Shell"
    +INVENTORY.UNDROPPABLE
    Weapon.YAdjust 11
	Inventory.RestrictedTo "BlakeStone"
    Tag "Slow Fire Protector"
    States
    {
      Spawn:
        BWP2 A -1
        Loop

      Ready:
		TNT1 A 0 A_JumpIfInventory("BlakeAOGToken", 1, 2)
        BPI2 A 1 A_WeaponReady
        loop
		BPI2 F 1 A_WeaponReady
		Loop

      Fire:
		TNT1 A 0 A_JumpIfInventory("BlakeAOGToken", 1, "FireAOG")
        TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"CoopFire")
        BPI2 B 3
        TNT1 A 0 A_PlayWeaponSound("Bpistol2/Fire")
		BPI2 C 3 BRIGHT A_FireBullets(1.8,1.8,1,21,"BlankPuffBlake")
      //BPI2 C 3 BRIGHT A_FireBullets(1.8,1.8,1,32,"BlankPuffBlake")
        BPI2 DEA 3
        Goto Ready
		
	  FireAOG:
        TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"CoopFireAOG")
        BPI2 G 3
        TNT1 A 0 A_PlayWeaponSound("Bpistol2/Fire")
		BPI2 H 3 BRIGHT A_FireBullets(1.8,1.8,1,21,"BlankPuffBlake")
      //BPI2 C 3 BRIGHT A_FireBullets(1.8,1.8,1,32,"BlankPuffBlake")
        BPI2 IJF 3
        Goto Ready	  

      CoopFire:
		BPI2 B 3
        TNT1 A 0 A_PlayWeaponSound("BPistol2/Fire")
        BPI2 C 3 BRIGHT A_FireBullets(1.8,1.8,1,21,"BlankPuffBlakeCoop")
        BPI2 DEA 3
        Goto Ready
		
	 CoopFireAOG:
        BPI2 G 3
        TNT1 A 0 A_PlayWeaponSound("Bpistol2/Fire")
		BPI2 H 3 BRIGHT A_FireBullets(1.8,1.8,1,21,"BlankPuffBlakeCoop")
      //BPI2 C 3 BRIGHT A_FireBullets(1.8,1.8,1,32,"BlankPuffBlake")
        BPI2 IJF 3
        Goto Ready	  
    }
}

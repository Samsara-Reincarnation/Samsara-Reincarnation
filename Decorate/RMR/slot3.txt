ACTOR "ACR ADD-ON" : Weapon
{
  Weapon.SelectionOrder 30
  Weapon.SlotNumber 3  
  Weapon.BobStyle Inverse
  Weapon.BobRangeX 0.85
  Weapon.BobSpeed 1.1  
  Weapon.AmmoUse 3
  Weapon.AmmoGive 18
  Weapon.AmmoType "Shell"
  Inventory.PickupSound "RMR/pickup" 	  
  Inventory.PickupMessage "You found the Recoiless ACL 3x Power Add-On!"
  Inventory.RestrictedTo "RebelMoonCommando"
  Tag "ACR Add-On"
  States
  {
  Ready:
    RM2X A 1 A_WeaponReady
    Loop
  Deselect:
	TNT1 A 0 A_Lower  
    RM2X A 1 A_Lower
	Loop
  Select:
	TNT1 A 0 A_Raise
    RM2X A 1 A_Raise
    Loop
  Fire:
    TNT1 A 0 A_PlaySound("RMR/ACRSHOT2") 	
	TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"ACRCoopFire")  
	TNT1 A 0 A_JumpIfInventory("DMModeOn",1,"ACRDMFire")  
	TNT1 A 0
	RM2X B 1 bright A_FireCustomMissile("RMRLaser3",0,2,0,0)
	TNT1 A 0 A_FireCustomMissile("RMRLaser3",0,0,8,0)
	TNT1 A 0 A_FireCustomMissile("RMRLaser3",0,0,-8,0)	
	RM2X CC 2 Bright	
	RM2X CD 2 Bright
	RM2X DD 2 Bright
    RM2X CB 2 Bright
    RM2X BA 3 Bright
    RM2X AA 1 Bright A_ReFire	
    Goto Ready
  ACRCoopFire:
	RM2X B 1 bright A_FireCustomMissile("RMRLaser3Coop",0,2,0,0)
	TNT1 A 0 A_FireCustomMissile("RMRLaser3Coop",0,0,8,0)
	TNT1 A 0 A_FireCustomMissile("RMRLaser3Coop",0,0,-8,0)	
	RM2X CC 2 Bright	
	RM2X CD 2 Bright
	RM2X DD 2 Bright
    RM2X CB 2 Bright
    RM2X BA 3 Bright
    RM2X AA 1 Bright A_ReFire	
    Goto Ready
  ACRDMFire:
	RM2X B 1 bright A_FireCustomMissile("RMRLaser3",0,2,0,0)
	TNT1 A 0 A_FireCustomMissile("RMRLaser3",0,0,8,0)
	TNT1 A 0 A_FireCustomMissile("RMRLaser3",0,0,-8,0)	
	RM2X CC 4 Bright	
	RM2X CD 4 Bright
	RM2X DD 4 Bright
    RM2X CB 4 Bright
    RM2X BA 4 Bright
    RM2X AA 3 Bright A_ReFire	
    Goto Ready
  Spawn:
    RM2X I -1 Bright
	Loop
  }
}

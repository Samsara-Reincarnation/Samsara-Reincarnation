ACTOR "Estron Ball" : Weapon
{
  weapon.selectionorder 240
  Weapon.SlotNumber 0 
  Inventory.PickupSound "RMR/pickup" 	  
  Inventory.PickupMessage "Estron Ball"
  +Weapon.NOAUTOAIM
  +INVENTORY.UNDROPPABLE  
  Obituary "%o didnt expect %k's snowballs."
  Inventory.RestrictedTo "RebelMoonCommando"
  Tag "Estron Ball"
  States
  {
  Ready:
    RM2X E 1 A_WeaponReady
    Loop
  Deselect:
    RM2X E 1 A_Lower
	Loop
  Select:
    RM2X E 1 A_Raise
    Loop
  Fire:
    TNT1 A 0 A_PlaySound("RMR/WPlasma")
	RM2X F 15 bright A_FireCustomMissile("RMR-WhitePlasma",0,1,0,0)
	RM2X G 15 bright
 	RM2X G 1 bright A_Refire
    Goto Ready
  Spawn:
	RM2X H -1
	Loop
  }
}

Actor RMR-WhitePlasma : FastProjectile
{
    Damage (10000)
    Radius 5
    Height 5
    Speed 80
    Projectile
	+BLOODLESSIMPACT	
	+FOILINVUL
	+EXTREMEDEATH
	DeathSound "RMR/WPlasmaB"
	Scale 0.50
    States
    {
      Spawn:
        RMZ3 MN 1 bright
        loop
      Death:
        RMZ3 OPQ 4 bright
		RMZ3 R 3 bright A_FadeOut
        stop
    }
}

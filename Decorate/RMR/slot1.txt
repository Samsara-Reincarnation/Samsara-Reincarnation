Actor "Dirtshark" : Weapon
{
    Weapon.Kickback 0
    Weapon.SelectionOrder 2200
	Weapon.BobStyle Inverse
	Weapon.BobRangeX 0.85
	Weapon.BobSpeed 1.1	
    Weapon.SlotNumber 1
    Weapon.SlotPriority 0	
    Inventory.PickupMessage "You got the MolyCarbide Power Mining Drill!"
    Inventory.PickupSound "RMR/pickup" 	
	Obituary "%k drilled %o a deep hole."
    +WEAPON.MELEEWEAPON
    +INVENTORY.UNDROPPABLE
    Inventory.RestrictedTo "RebelMoonCommando"
    Tag "\"Dirtshark\" MolyCarbide Power Mining Drill"
    States
    {
      Ready:
        RM0G A 4 A_WeaponReady
        loop
      Deselect:
		TNT1 A 0 A_Lower	  
        RM0G A 1 A_Lower
        loop
      Select:
		TNT1 A 0 A_Raise	  
        RM0G A 1 A_Raise
        loop
      Fire:
		RM0G BC 4 BRIGHT
        RM0G D 0 A_JumpIfInventory("CoopModeOn",1,2)
		RM0G D 4 BRIGHT A_Saw("rmr/drillfull", "rmr/drillhit", 30, "RMRPuff", SF_NORANDOM)
        goto FireEnd
		RM0G D 4 BRIGHT A_Saw("rmr/drillfull", "rmr/drillhit", 30, "RMRPuffCoop", SF_NORANDOM)
        goto FireEnd
	  FireEnd:
		RM0G A 0 A_ReFire		
        goto Ready 
      Spawn:
        NULL A -1
        stop
    }
}

ACTOR CMSMG : Weapon
{
  obituary "%o was peppered by %k's SMG."
  inventory.pickupmessage "You got the SMG!"
  Weapon.AmmoType "Clip" 
  weapon.ammouse 1
  Weapon.SelectionOrder 3200  
  weapon.ammogive 40
  Weapon.BobStyle InverseAlpha
  Weapon.BobRangeX 0.5
  Weapon.BobRangeY 0.6
  Weapon.BobSpeed 2.3  
  +FLOORCLIP
  +Inventory.UNDROPPABLE
  Decal "BulletChip"
  Weapon.SlotNumber 4
  Weapon.YAdjust 9 
  Inventory.RestrictedTo "CMPlayer"  
  Tag "SMG"  
  states
  {
  Spawn:
    CMMG F -1
    stop
  Ready:
    CMMG A 1 A_WeaponReady
    loop
  Deselect: 
    TNT1 A 0 A_Lower
    CMMG G 1 A_Lower
    loop
  Select:
    TNT1 A 0 A_Raise
    CMMG G 1 A_Raise
    loop
  Fire:
    CMMG B 0 A_Jump(128,"Fire2") 
	TNT1 A 0 A_PlaySound("CM/SMGFire",CHAN_WEAPON)
	TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"CoopFire")	
    CMMG C 3 bright A_FireCustomMissile("CybermageBullet",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG D 3 bright A_FireCustomMissile("CybermageBullet",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG E 3 bright A_FireCustomMissile("CybermageBullet",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CHGG B 0 A_ReFire
	goto Ready
  Fire2:	
	TNT1 A 0 A_PlaySound("CM/SMGFire",CHAN_WEAPON)
	TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"CoopFire2")		
    CMMG H 3 bright A_FireCustomMissile("CybermageBullet",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG I 3 bright A_FireCustomMissile("CybermageBullet",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG J 3 bright A_FireCustomMissile("CybermageBullet",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CHGG B 0 A_ReFire
    goto Ready
  CoopFire:
    CMMG C 3 bright A_FireCustomMissile("CybermageBulletCoop",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG D 3 bright A_FireCustomMissile("CybermageBulletCoop",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG E 3 bright A_FireCustomMissile("CybermageBulletCoop",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CHGG B 0 A_ReFire
	Goto Ready
  CoopFire2:
    CMMG H 3 bright A_FireCustomMissile("CybermageBulletCoop",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG I 3 bright A_FireCustomMissile("CybermageBulletCoop",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CMMG J 3 bright A_FireCustomMissile("CybermageBulletCoop",fRandom(6.00,-6.00),1,0,fRandom(-4.00,4.00))
    CHGG B 0 A_ReFire
    goto Ready 
  }
}
